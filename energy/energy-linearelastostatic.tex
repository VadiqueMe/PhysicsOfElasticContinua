\begin{equation*}\begin{array}{c}
\variation{\hspace{.1ex}\bigl( \hspace{-0.2ex} \linearstress \dotdotp \mathboldepsilon \bigr)} \hspace{-0.2ex}
= \linearstress \dotdotp \variation{\mathboldepsilon} + \variation{\hspace{-0.1ex}\linearstress} \dotdotp \mathboldepsilon
= \variation{\Pi} + \variation{\widehat{\Pi}}
\\[.3em]
%
\variation{\Pi} = \linearstress \dotdotp \variation{\mathboldepsilon}
= \mathboldepsilon \hspace{-0.1ex} \dotdotp \hspace{-0.1ex} \stiffnesstensor \hspace{-0.1ex} \dotdotp \variation{\mathboldepsilon} ,
\:\:
\variation{\widehat{\Pi}} = \variation{\hspace{-0.1ex}\linearstress} \dotdotp \mathboldepsilon
= \linearstress \hspace{-0.1ex} \dotdotp \hspace{-0.2ex} \pliabilitytensor \hspace{-0.1ex} \dotdotp \variation{\hspace{-0.1ex}\linearstress}
\end{array}\end{equation*}
