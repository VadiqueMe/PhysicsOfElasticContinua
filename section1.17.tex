\en{The~}\inquotes{\ru{Правило произведения~(}product rule\ru{)}}\en{ is}\ru{---}
\en{about}\ru{про}
\en{the~differentiation}\ru{дифференцирование}
\en{of~}${p \hspace{-0.2ex} \circ \hspace{-0.2ex} q}$,
\en{the~product}\ru{произведения}
\en{of~}$p$ \en{and}\ru{и}~$q$\::

\nopagebreak\vspace{-0.4em}
\begin{equation*}
d \bigl( \hspace{.15ex} p \circ q \hspace{.1ex} \bigr)
\hspace{-0.1ex} = \hspace{-0.1ex}
\bigl( p + dp \bigr) \hspace{-0.4ex} \circ \hspace{-0.4ex} \bigl( q + dq \bigr) \hspace{-0.2ex} - \hspace{.1ex} p \circ q
\hspace{.1ex} = \hspace{.1ex}
dp \circ q \hspace{.2ex} + \hspace{.2ex} p \circ dq \hspace{.2ex} + \hspace{.2ex} dp \circ dq
\hspace{.2ex} ,
\end{equation*}

\vspace{-0.5em}\noindent
\en{where}\ru{где}
${dp \circ dq = \hspace{-0.2ex} \infinitesimalof{\bigl(dp, dq\bigr)}}$\:---
\en{the~product}\ru{произведение}~${dp \hspace{-0.2ex} \circ \hspace{-0.2ex} dq}$
\en{of two infinitesimals}\ru{двух бесконечном\'{а}лых}
\en{is~considered to be}\ru{считается}
\en{infinitesimally smaller}\ru{бесконечном\'{а}ло м\'{е}ньшим}\ru{,}
\en{than}\ru{чем}
\en{either}\ru{каждое из}
\en{infinitesimal}\ru{бесконечном\'{а}лых}~(${dp}$ \en{or}\ru{или}~${dq}$)
\en{alone}\ru{в~одиночку}.
%
\en{Thus}\ru{Так что},
\en{the~term}\ru{член}
${dp \hspace{-0.2ex} \circ \hspace{-0.2ex} dq}$
\en{can be}\ru{может быть}
\en{dropped}\ru{отброшен}
\en{as negligible}\ru{как пренебрежимый}
\en{compared to terms}\ru{по~сравнению с~членами}
\en{with one infinitesimal}\ru{с~одной бесконечном\'{а}лой}.
%

\begin{equation}\label{theproductrule.differentiation}
