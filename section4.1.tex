$\volumeloadvector$ \en{and}\ru{и}~$\bm{p}$
\en{increase}\ru{растут}
\en{by}\ru{в}~$m$
\en{times}\ru{раз}
(\en{the~body}\ru{тело}
\en{is fixed on}\ru{закреплено на}~${o_1}$),
\en{then}\ru{то}
$\bm{u}$,
$\infinitesimaldeformation$
\en{and}\ru{и}~$\linearstress$
\en{will increase}\ru{вырастут}
\ru{тоже }\en{by}\ru{в}~$m$ \en{times}\ru{раз}\en{ too}.
\en{The~potential energy}\ru{Потенциальная энергия}~$\potentialenergydensity$
\en{will increase by}\ru{вырастет в}~$m^2$ \en{times}\ru{раз}.
\en{In~reality}\ru{В~реальности},
\en{this}\ru{это}
\en{is observed}\ru{наблюдается}
\en{only}\ru{лишь}
\en{when}\ru{когда}
\en{the~loads}\ru{нагрузки}
\en{are small}\ru{малые}.

\subsection*{\en{The potential energy}\ru{Потенциальная энергия}}

\en{With linearity}\ru{С~линейностью}\en{,}
\en{the~potential energy density}\ru{плотность потенциальной энергии}~$\potentialenergydensity$
\en{as a~function}\ru{как функция}
\en{of infinitesimal deformation}\ru{бесконечномалой деформации}~$\infinitesimaldeformation$
\en{is}\ru{есть}
\en{a~}\href{https://encyclopediaofmath.org/wiki/Quadratic_form}{\en{quadratic form}\ru{квадратичная форма}}

\nopagebreak\vspace{-0.1em}
\begin{equation}\label{linear.potentialenergydensity.quadratic}
\potentialenergydensity ( \hspace{-0.1ex}
\infinitesimaldeformation
\hspace{-0.1ex} ) \hspace{-0.2ex}
= \hspace{.1ex}
\smash{\smalldisplaystyleonehalf} \hspace{.3ex}
\infinitesimaldeformation
\hspace{-0.1ex} \dotdotp \hspace{-0.1ex}
\stiffnesstensor
\dotdotp \hspace{-0.1ex}
\infinitesimaldeformation
\hspace{.2ex} ,
\end{equation}

\vspace{-0.5em}\noindent
\en{the~variation of~which is}\ru{вариация которой}

\nopagebreak\vspace{-0.2em}
\begin{equation*}
\variation{\potentialenergydensity}
= \hspace{.1ex}
\smash{\smalldisplaystyleonehalf} \hspace{.25ex}
\variation{ \bigl(
   \infinitesimaldeformation
   \hspace{-0.1ex} \dotdotp \hspace{-0.1ex}
   \stiffnesstensor
   \dotdotp \hspace{-0.1ex}
   \infinitesimaldeformation
\bigr)
} \hspace{-0.25ex}
= \hspace{.1ex}
\smash{\smalldisplaystyleonehalf}
\hspace{-0.1ex} \bigl( \hspace{.2ex}
\scalebox{.95}{$
   \variation{\infinitesimaldeformation}
   \hspace{-0.1ex} \dotdotp \hspace{-0.1ex}
   \stiffnesstensor
   \dotdotp \hspace{-0.1ex}
   \infinitesimaldeformation
   \hspace{.1ex} + \hspace{.1ex}
   \infinitesimaldeformation
   \hspace{-0.1ex} \dotdotp \hspace{-0.1ex}
   \stiffnesstensor
   \dotdotp
   \variation{\infinitesimaldeformation}
$}
\bigr)
\hspace{-0.25ex} =
\tikzmark{beginLinearStressAsDeformationAndStiffness}
\infinitesimaldeformation
\hspace{-0.1ex} \dotdotp \hspace{-0.1ex}
\stiffnesstensor
\tikzmark{endLinearStressAsDeformationAndStiffness}
\dotdotp
\variation{\infinitesimaldeformation}
%%\hspace{.2ex} .
\end{equation*}%
\AddUnderBrace[line width=.75pt][0, -0.33ex][yshift = -0.1ex]%
{beginLinearStressAsDeformationAndStiffness}{endLinearStressAsDeformationAndStiffness}%
{${
   \scalebox{.8}{$ \scalebox{.83}{$
      \displaystyle\frac{\raisemath{-0.16em}{\partial \hspace{.1ex} \potentialenergydensity}}{\partial \infinitesimaldeformation}
   $} \hspace{-0.3ex} = \hspace{-0.3ex} \linearstress $}
}$}
\vspace{-1.2em}

\noindent
\en{and the~second variation}\ru{и~вторая вариация}

\nopagebreak\vspace{-0.2em}\begin{equation*}
\secondvariation{\hspace{.1ex}\potentialenergydensity}
= \variation{\infinitesimaldeformation}
\hspace{-0.1ex} \dotdotp
\stiffnesstensor
\dotdotp
\variation{\infinitesimaldeformation}
\hspace{.2ex} .
\end{equation*}

\vspace{-0.4em}\noindent
\en{The relation}\ru{Отношение}
\en{between}\ru{между}
\en{the~stiffness tensor}\ru{тензором жёсткости}
\en{and }\ru{и~}%
\en{the~potential energy}\ru{потенциальной энергией}
\en{of~elastic deformation}\ru{упругой деформации}
\en{is now clear}\ru{теперь ясн\'{а}}\::

\nopagebreak\vspace{-0.2em}
\begin{equation*}
   \stiffnesstensor
   \hspace{.1ex} =
   \scalebox{.9}{$ \displaystyle\frac{
      \raisemath{-0.1em}{ \partial^2 \hspace{.1ex} \potentialenergydensity }
   }{
      \raisemath{-0.1em}{
         \partial \infinitesimaldeformation \hspace{.1ex}
         \partial \infinitesimaldeformation
      }
   } $}
   \hspace{.2ex} ,
\end{equation*}

\vspace{-0.3em}\noindent
\en{because}\ru{потому что}
%
${ \variation{\hspace{.1ex}\potentialenergydensity}
( \hspace{-0.1ex}
    \infinitesimaldeformation
\hspace{-0.1ex} ) \hspace{-0.2ex}
= \smash{\scalebox{.9}{$
    \displaystyle\frac{
        \raisemath{-0.16em}
        {\partial \hspace{.1ex} \potentialenergydensity}}
        {\partial \infinitesimaldeformation}
    $}} \hspace{-0.1ex}
\dotdotp
\variation{\infinitesimaldeformation} }$
%
\en{and}\ru{и}~%
%
${ \secondvariation{\hspace{.1ex}\potentialenergydensity}(\hspace{-0.1ex}
\infinitesimaldeformation
\hspace{-0.1ex}) \hspace{-0.2ex}
= \variation{\infinitesimaldeformation}
\hspace{-0.1ex} \dotdotp
\smash{\scalebox{.9}{$ \displaystyle\frac{\raisemath{-0.16em}{\partial^2 \hspace{.1ex} \potentialenergydensity}}{ \partial \infinitesimaldeformation \hspace{.1ex} \partial \infinitesimaldeformation } $}}
\hspace{-0.1ex} \dotdotp
\variation{\infinitesimaldeformation}\hspace{.1ex} }$.

\vspace{.3em}
\en{Adding that here}\ru{Добавляя что тут},
\en{as well as for}\ru{так~же, как~и для}
\en{1\hbox{-}variable calculus}\ru{исчисления функций одной переменной}\footnote{%
\en{for}\ru{для}\hspace{.5ex}~%
${ y(x) = \hspace{.1ex} \smallerdisplaystyleonehalf \upalpha \hspace{.2ex} x^2 \hspace{-0.2ex} = \hspace{.1ex} \smallerdisplaystyleonehalf x \upalpha \hspace{.2ex} x }$
\hspace{1em}\begin{minipage}[t]{.5\textwidth}
${ dy = \hspace{.1ex} \smallerdisplaystyleonehalf \bigl( dx \upalpha \hspace{.2ex} x + x \upalpha \hspace{.2ex} dx \bigr) \hspace{-0.2ex} = x \upalpha \hspace{.2ex} dx }$
\\
${ d^2 y = dx \upalpha \hspace{.2ex} dx = \upalpha \hspace{.1ex} (dx)^2 }$
\end{minipage}
\\[-0.5em]
\en{that is}\ru{то есть}\hspace{.5ex}~%
${ d^2 y = 2 \hspace{.2ex} y(dx) }$
}\hbox{\hspace{-0.6ex},}
${
\secondvariation{\hspace{.1ex}\potentialenergydensity} = 2 \hspace{.25ex} \potentialenergydensity (\variation{\infinitesimaldeformation})
}$,
\en{the~first and~second}\ru{первая и~вторая}
\en{variations of}\ru{вариации}~$\potentialenergydensity$
\en{may~be written as}\ru{могут быть записаны как}

\nopagebreak\vspace{-0.2em}
\begin{equation}\label{thefirstandsecondvariationsofpotentialenergydensity.linearelasticity}
\begin{gathered}
\variation{\hspace{.1ex}\potentialenergydensity}
( \hspace{-0.1ex}
    \infinitesimaldeformation
\hspace{-0.1ex} ) \hspace{-0.2ex}
= \scalebox{.9}{$
    \displaystyle\frac{
        \raisemath{-0.16em}
        {\partial \hspace{.1ex} \potentialenergydensity}}
        {\partial \infinitesimaldeformation}
    $} \hspace{-0.1ex}
\dotdotp
\variation{\infinitesimaldeformation}
= \infinitesimaldeformation
\hspace{-0.1ex} \dotdotp \hspace{-0.1ex}
\stiffnesstensor
\dotdotp
\variation{\infinitesimaldeformation}
= \linearstress
\dotdotp
\variation{\infinitesimaldeformation}
\hspace{.2ex} ,
\\
%
\secondvariation{\hspace{.1ex}\potentialenergydensity}(\hspace{-0.1ex}
\infinitesimaldeformation
\hspace{-0.1ex}) \hspace{-0.2ex}
= \variation{\infinitesimaldeformation}
\hspace{-0.1ex} \dotdotp
\scalebox{.9}{$ \displaystyle\frac{\raisemath{-0.16em}{\partial^2 \hspace{.1ex} \potentialenergydensity}}{ \partial \infinitesimaldeformation \hspace{.1ex} \partial \infinitesimaldeformation } $}
\hspace{-0.1ex} \dotdotp
\variation{\infinitesimaldeformation}
= \variation{\infinitesimaldeformation}
\hspace{-0.1ex} \dotdotp
\stiffnesstensor
\dotdotp
\variation{\infinitesimaldeformation}
= 2 \hspace{.25ex} \potentialenergydensity (\variation{\infinitesimaldeformation})
\hspace{.1ex} .
\end{gathered}
\end{equation}

\subsection*{\en{The principle of virtual work}\ru{Принцип виртуальной работы}}

\ru{Принцип}\en{The} \hbox{d’\hspace{-0.2ex}Alembert\ru{’а}--Lagrange\ru{’а}}\en{ principle}
\en{of~virtual work}\ru{виртуальной работы}~%
\eqrefwithchapterdotsection{discrete:principleofvirtualwork}{chapter:classicalmechanics}{section:virtualworkprinciple.classicalmechanics},
\en{which}\ru{который}
\en{can be used}\ru{может быть использован}
\en{as the~foundation}\ru{как фундамент}
\en{of~mechanics}\ru{механики},
\en{applies}\ru{примен\'{и}м}
\en{to the~linear theory as well}\ru{и~к~линейной теории тоже}.
%
\en{Since}\ru{Поскольку}
\en{internal forces}\ru{внутренние силы}
\en{in any elastic medium}\ru{в~любой упругой среде}
\en{are potential}\ru{потенциальны}
(${ \variation{\internalwork} = - \hspace{.2ex} \variation{\potentialenergydensity} }$),
\en{the~principle}\ru{принцип}
\en{is formulated}\ru{формулируется}
\en{as}\ru{как}

\nopagebreak\vspace{-0.4em}
\begin{equation}\label{lineartheory:principleofvirtualwork}
\displaystyle\integral\displaylimits_{\mathcal{V}}
\hspace{-0.2ex} \Bigl( \hspace{-0.1ex}
    \bigl(
        \hspace{.1ex}
        \volumeloadvector
        \hspace{-0.1ex} - \hspace{-0.1ex}
        \massdensity \hspace{.1ex}
        \mathdotdotabove{\bm{u}}
        \hspace{.2ex}
    \bigr)
    \hspace{-0.3ex}
    \dotp
    \variation{\bm{u}} - \variation{\potentialenergydensity}
\hspace{.1ex} \Bigr) d\mathcal{V}
\hspace{.1ex} + \hspace{-0.3ex}
\displaystyle\integral\displaylimits_{o_2} \hspace{-0.4ex} \bm{p} \dotp \variation{\bm{u}} \hspace{.25ex} do \hspace{.1ex}
= \hspace{.1ex} 0
\hspace{.1ex} ,
\hspace{.8em}
%
\bm{u} \hspace{.1ex}
\bigr|_{o_1}
\hspace{-0.64ex} = \hspace{.1ex}
\zerovector
\hspace{.2ex} .
\vspace{-0.2em}
\end{equation}

\vspace{-0.2em}\noindent
\en{In~addition}\ru{Вдобавок},
\en{if the~medium is elastic}\ru{если среда упруга}
\emph{\en{linearly}\ru{линейно}},
\en{then}\ru{то}

\nopagebreak\vspace{-0.4em}\begin{equation*}
\begin{gathered}
\variation{\infinitesimaldeformation} \hspace{-0.1ex}
= \hspace{.1ex} \variation{\bigl( \hspace{.1ex} \boldnabla \bm{u}^{\hspace{.1ex}\mathsf{S}} \hspace{.2ex} \bigr)} \hspace{-0.2ex}
= \hspace{-0.2ex} \boldnabla \bigl( \hspace{.1ex} \variation{\bm{u}} \bigr)^{\hspace{-0.2ex}\mathsf{S}}
\hspace{.5em} \text{\en{or just}\ru{или просто}} \hspace{.4em}
\boldnabla \variation{\bm{u}}^{\hspace{.1ex}\mathsf{S}}
\hspace{-0.1ex} ,
\\[.4em]
%
\variation{\potentialenergydensity}
= \linearstress \dotdotp \variation{\infinitesimaldeformation}
= \linearstress \dotdotp \hspace{-0.12ex} \boldnabla \variation{\bm{u}}^{\hspace{.1ex}\mathsf{S}} \hspace{-0.16ex}
= \boldnabla \hspace{-0.1ex} \dotp \left( \hspace{.1ex} \linearstress \hspace{-0.1ex} \dotp \variation{\bm{u}} \hspace{.1ex} \right) - \boldnabla \dotp \linearstress \dotp \variation{\bm{u}}
\hspace{.1ex} ,
\\[.4em]
%
\displaystyle \integral\displaylimits_{\mathcal{V}} \hspace{-0.5ex} \variation{\potentialenergydensity} \hspace{.2ex} d\mathcal{V} =
\ointegral\displaylimits_{\mathclap{o\hspace{.1ex}(\boundary \mathcal{V})}} \hspace{-0.2ex} \unitnormalvector \hspace{.1ex} \dotp \linearstress \dotp \variation{\bm{u}} \hspace{.32ex} do \hspace{.2ex} - \hspace{-0.1ex}
\integral\displaylimits_{\mathcal{V}} \hspace{-0.5ex} \boldnabla \dotp \linearstress \dotp \variation{\bm{u}} \hspace{.32ex} d\mathcal{V}
\hspace{-0.1ex} ,
\end{gathered}
\end{equation*}

\vspace{-0.4em}\noindent
\en{and}\ru{и}~\eqref{lineartheory:principleofvirtualwork}
\en{becomes}\ru{становится}

\nopagebreak\vspace{-0.4em}
\begin{equation*}
\displaystyle\integral\displaylimits_{\mathcal{V}} \hspace{-0.4ex} \Bigl( \hspace{-0.1ex} \boldnabla \dotp \linearstress \hspace{.1ex} + \volumeloadvector \hspace{-0.1ex} - \hspace{-0.1ex} \massdensity \hspace{.1ex} \mathdotdotabove{\bm{u}} \hspace{.1ex} \Bigr) \hspace{-0.3ex} \dotp \variation{\bm{u}} \hspace{.25ex} d\mathcal{V}
+ \hspace{-0.3ex}
\displaystyle\integral\displaylimits_{o_2} \hspace{-0.4ex} \Bigl( \bm{p} - \unitnormalvector \dotp \linearstress \Bigr) \hspace{-0.3ex} \dotp \variation{\bm{u}} \hspace{.25ex} do
\hspace{.2ex} = \hspace{.2ex} 0
\hspace{.2ex} .
\end{equation*}

\vspace{-0.4em}\noindent
\en{Here}\ru{Здесь}\en{,}
\en{virtual displacements}\ru{виртуальные смещения}~$\variation{\bm{u}}$
\en{are compatible}\ru{совместимы}
\en{with }\ru{с~}\en{the boundary condition}\ru{краевым условием}
\en{for displacements}\ru{для смещений}
\en{in}\ru{в}~\eqref{lineartheory:principleofvirtualwork},
${ \variation{\bm{u}} \hspace{.1ex} \bigr|_{o_1} \hspace{-0.64ex} = \hspace{.2ex} \zerovector }$.
